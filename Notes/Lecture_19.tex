\section{The definition of Eigenvalues}
\begin{definition}
    A \textbf{linear transformation} is a function $T: V \to W$ such that 
    \begin{enumerate}
        \item $T(u + v) = T(u) + T(v)$ for all $u, v \in V$
        \item $T(cu) = cT(u)$ for all $u \in V$ and $c \in \mathbb{R}$
    \end{enumerate}
\end{definition}
